	2.0.0p247 :001 > a = %q{helo world\n}
	 => "helo world\\n" 
	2.0.0p247 :002 > b=%Q{Hello World\n}
	 => "Hello World\n" 
	2.0.0p247 :003 > puts a
	helo world\n
	 => nil 
	2.0.0p247 :004 > puts b
	Hello World
	 => nil
